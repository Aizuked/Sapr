public class Main {
    public static void main ( String [ ] args ) {
        int k = 3 ;
        float result = 0 ;
        for ( int i = 0 ; i < 10 ; i ++ ) {
            System.out.println ( i ) ;

            for ( int j = 1 ; j < k ; j ++ ) {
                result += 0.333 ;
            }
        }
    }
}
